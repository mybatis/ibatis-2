32  /**
33   * Regression test for JIRA IBATIS-260: "Hash conflict with groupBy resultMaps".
34   * <p>
35   * The JIRA was about a problem using groupBy in a hierarchy (look in the SqlMap of this test case for an example). The
36   * problem was that when a value of a groupBy key for the second level groupBy existed in more than 1 first level
37   * groupBy some objects would be assigned to the wrong 2nd level group.
38   * <p>
39   * The reason for this was that the values of the keys of the groupBys would only take into account the current groupBy
40   * level. The fix is to keep track of all of the values for the keys used in previous groupBys in the hierarchy so that
41   * the objects can be properly assigned.
42   * <p>
43   * Broken in v2.1.7, v2.1.6, and earlier. Fixed in versions > 2.1.7
44   *
45   * @author Sven Boden
46   */
47  class JiraIbatis260Test extends BaseSqlMap {
48  
49    @BeforeEach
50    void setUp() throws Exception {
51      initSqlMap("com/ibatis/sqlmap/maps/SqlMapConfig.xml", null);
52      initScript("scripts/jira.sql");
53    }
54  
55    /**
56     * Regression test case for JIRA IBATIS-260. This is also a good example of the power of groupBy. We load the exact
57     * same data but once via groupBy and once via a flat structure. Then we put the grouped result in a hashMap and we
58     * delete all entries from the flat result from that hashMap. If we don't find a key in the hashMap there's a problem,
59     * if at the end the hashMap is not empty there's also a problem.
60     *
61     * @throws Exception
62     *           none should be thrown (if the regression test succeeds)
63     */
64    @Test
65    void testIbatis260Error1() throws Exception {
66      List<?> groupedResult = sqlMap.queryForList("getJira260GroupedResult", null);
67  
68      HashMap<String, Object> test = new HashMap<>();
69      Iterator<?> indexIterator = groupedResult.iterator();
70      while (indexIterator.hasNext()) {
71        ArticleIndex articleIndex = (ArticleIndex) indexIterator.next();
72        Iterator<?> topicIterator = articleIndex.getTopics().iterator();
73        while (topicIterator.hasNext()) {
74          Topic topic = (Topic) topicIterator.next();
75          Iterator<?> descriptionIterator = topic.getDescriptionList().iterator();
76          while (descriptionIterator.hasNext()) {
77            TopicDescription desc = (TopicDescription) descriptionIterator.next();
78  
79            // Put a flattened key in the hashMap
80            test.put(articleIndex.getCategoryTitle() + "||" + topic.getTopicTitle() + "||" + desc.getDescription(), null);
81          }
82        }
83      }
84  
85      // Iterate over the flat version of the results and remove
86      // all those keys from the hashMap. If an entry does not exists
87      // the test case fails. If at the end the hashMap is not empty
88      // the test case also fails.
89      String key = null;
90      List<?> flatResult = sqlMap.queryForList("getJira260FlatResult", null);
91      Iterator<?> iterator = flatResult.iterator();
92      while (iterator.hasNext()) {
93        ArticleIndexDenorm articleIndex = (ArticleIndexDenorm) iterator.next();
94        key = articleIndex.getCategoryTitle() + "||" + articleIndex.getTopicTitle() + "||"
95            + articleIndex.getDescription();
96        if (!test.containsKey(key)) {
97          throw new Exception("Key \"" + key + "\" does not exist in the hashMap, regression test fails");
98        } else {
99          test.remove(key);
100       }
101     }
102 
103     if (test.size() != 0) {
104       throw new Exception("Map is not empty at the end of the test, regression test fails");
105     }
106   }
107 }
