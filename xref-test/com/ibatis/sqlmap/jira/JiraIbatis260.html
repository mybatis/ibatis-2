18  import java.util.HashMap;
19  import java.util.Iterator;
20  import java.util.List;
21  
22  import testdomain.ArticleIndex;
23  import testdomain.ArticleIndexDenorm;
24  import testdomain.Topic;
25  import testdomain.TopicDescription;

27  import com.ibatis.sqlmap.BaseSqlMapTest;

29  /**
30   * Regression test for JIRA IBATIS-260: "Hash conflict with groupBy resultMaps".
31   * <p>
32   * The JIRA was about a problem using groupBy in a hierarchy (look in the SqlMap of this test case for an example). The
33   * problem was that when a value of a groupBy key for the second level groupBy existed in more than 1 first level
34   * groupBy some objects would be assigned to the wrong 2nd level group.
35   * <p>
36   * The reason for this was that the values of the keys of the groupBys would only take into account the current groupBy
37   * level. The fix is to keep track of all of the values for the keys used in previous groupBys in the hierarchy so that
38   * the objects can be properly assigned.
39   * <p>
40   * Broken in v2.1.7, v2.1.6, and earlier. Fixed in versions > 2.1.7
41   * 
42   * @author Sven Boden
43   * 
44   */
45  public class JiraIbatis260 extends BaseSqlMapTest {
46  
47    @Override
48    protected void setUp() throws Exception {
49      initSqlMap("com/ibatis/sqlmap/maps/SqlMapConfig.xml", null);
50      initScript("scripts/jira.sql");
51    }
52  
53    /**
54     * Regression test case for JIRA IBATIS-260. This is also a good example of the power of groupBy. We load the exact
55     * same data but once via groupBy and once via a flat structure. Then we put the grouped result in a hashMap and we
56     * delete all entries from the flat result from that hashMap. If we don't find a key in the hashMap there's a problem,
57     * if at the end the hashMap is not empty there's also a problem.
58     * 
59     * @throws Exception
60     *           none should be thrown (if the regression test succeeds)
61     */
62    public void testIbatis260Error1() throws Exception {
63      List groupedResult = sqlMap.queryForList("getJira260GroupedResult", null);
64  
65      HashMap test = new HashMap();
66      Iterator indexIterator = groupedResult.iterator();
67      while (indexIterator.hasNext()) {
68        ArticleIndex articleIndex = (ArticleIndex) indexIterator.next();
69        Iterator topicIterator = articleIndex.getTopics().iterator();
70        while (topicIterator.hasNext()) {
71          Topic topic = (Topic) topicIterator.next();
72          Iterator descriptionIterator = topic.getDescriptionList().iterator();
73          while (descriptionIterator.hasNext()) {
74            TopicDescription desc = (TopicDescription) descriptionIterator.next();
75  
76            // Put a flattened key in the hashMap
77            test.put(articleIndex.getCategoryTitle() + "||" + topic.getTopicTitle() + "||" + desc.getDescription(), null);
78          }
79        }
80      }
81  
82      // Iterate over the flat version of the results and remove
83      // all those keys from the hashMap. If an entry does not exists
84      // the test case fails. If at the end the hashMap is not empty
85      // the test case also fails.
86      String key = null;
87      List flatResult = sqlMap.queryForList("getJira260FlatResult", null);
88      Iterator iterator = flatResult.iterator();
89      while (iterator.hasNext()) {
90        ArticleIndexDenorm articleIndex = (ArticleIndexDenorm) iterator.next();
91        key = articleIndex.getCategoryTitle() + "||" + articleIndex.getTopicTitle() + "||"
92            + articleIndex.getDescription();
93        if (!test.containsKey(key)) {
94          throw new Exception("Key \"" + key + "\" does not exist in the hashMap, regression test fails");
95        } else {
96          test.remove(key);
97        }
98      }
99  
100     if (test.size() != 0) {
101       throw new Exception("Map is not empty at the end of the test, regression test fails");
102     }
103   }
104 }
