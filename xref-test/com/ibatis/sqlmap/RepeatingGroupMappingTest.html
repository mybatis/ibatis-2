49    @Test
50    void testNestedProperties() throws Exception {
51      List<?> list = sqlMap.queryForList("getFish", null);
52      assertEquals(1, list.size());
53  
54      Category cat = (Category) list.get(0);
55      assertEquals("FISH", cat.getCategoryId());
56      assertEquals("Fish", cat.getName());
57      assertNotNull(cat.getProductList(), "Expected product list.");
58      assertEquals(4, cat.getProductList().size());
59  
60      Product product = (Product) cat.getProductList().get(0);
61      assertEquals(2, product.getItemList().size());
62    }
63  
64    /**
65     * This is a test case for iBATIS JIRA-250 "SELECT statement returns unexpected result when 'groupBy' and 'nullValue'
66     * are specified in resultMaps."
67     * <p>
68     * The problem was that when a child object in a resultmap only contained null values it would still be created when
69     * one of the properties of the child object contained a nullValue attribute in the ResultMap. The nullValue would be
70     * applied before checking whether all properties of the child were 'null', so at least 1 property would always be
71     * non-null (the reason for the nullValue attribute).
72     * <p>
73     * The fix is to first check whether all properties of the child are 'null', and if the child object contains at least
74     * 1 non-null property to then only create the child object and apply the nullValue attribute.
75     *
76     * @throws Exception
77     *           none should be thrown.
78     */
79    @Test
80    void testGroupByJIRA250() throws Exception {
81      List<?> list = sqlMap.queryForList("getAllProductCategoriesJIRA250", null);
82      Category cat = (Category) list.get(0);
83      assertEquals(0, cat.getProductList().size());
84    }
85  }
