16  package com.ibatis.sqlmap.engine.mapping.result.loader;
17  
18  import static org.junit.jupiter.api.Assertions.assertEquals;
19  import static org.junit.jupiter.api.Assertions.fail;
20  
21  import com.ibatis.sqlmap.engine.impl.SqlMapClientImpl;
22  import com.ibatis.sqlmap.engine.impl.SqlMapExecutorDelegate;
23  import com.ibatis.sqlmap.engine.mapping.result.loader.test.Bean1;
24  import com.ibatis.sqlmap.engine.mapping.result.loader.test.Bean2;
25  
26  import java.sql.SQLException;
27  
28  import org.junit.jupiter.api.Test;
29  
30  class EnhancedLazyResultLoaderTest {
31  
32    /**
33     * Test if a method in a proxied object can access a default method of another class in the same package with default
34     * access modifier.
35     * <p>
36     * Depending of the implementation of the Cglib-proxy the access will throw an IllegalAccessException.
37     */
38    @Test
39    void testProxyMethodAccess() throws SQLException {
40      SqlMapClientImpl client = setupMockSqlMapClientImpl();
41  
42      EnhancedLazyResultLoader loader = new EnhancedLazyResultLoader(client, "bean2", null, Bean2.class);
43      Bean2 bean2 = (Bean2) loader.loadResult();
44  
45      // cglib might throw an IllegalAccessException
46      bean2.testDefaultAccess();
47    }
48  
49    /**
50     * Test if a proxy for a null nevertheless dispatch to a default object.
51     */
52    @Test
53    void testNullProxy() throws SQLException {
54      SqlMapClientImpl client = setupMockSqlMapClientImpl();
55  
56      EnhancedLazyResultLoader loader = new EnhancedLazyResultLoader(client, "bean3", null, TestBean3.class);
57      TestBean3 bean = (TestBean3) loader.loadResult();
58  
59      assertEquals("foobar", bean.getText());
60    }
61  
62    /**
63     * Mock a {@link SqlMapClientImpl} which returns a {@link Bean1} for the id <code>bean1</code> and {@link Bean2} for
64     * id <code>bean2</code>.
65     */
66    private SqlMapClientImpl setupMockSqlMapClientImpl() {
67      SqlMapExecutorDelegate delegate = new SqlMapExecutorDelegate();
68      SqlMapClientImpl client = new SqlMapClientImpl(delegate) {
69        @Override
70        public Object queryForObject(String id, Object paramObject) throws SQLException {
71          if ("bean1".equals(id)) {
72            return new Bean1();
73          } else if ("bean2".equals(id)) {
74            Bean2 bean = new Bean2();
75            EnhancedLazyResultLoader loader = new EnhancedLazyResultLoader(this, "bean1", null, Bean1.class);
76            bean.setBean1((Bean1) loader.loadResult());
77            return bean;
78          } else if ("bean3".equals(id)) {
79            return null;
80          }
81          fail();
82          return null;
83        }
84      };
85      return client;
86    }
87  
88    public static class TestBean3 {
89  
90      public String getText() {
91        return "foobar";
92      }
93  
94    }
95  }
