18  import static org.junit.jupiter.api.Assertions.assertEquals;
19  import static org.junit.jupiter.api.Assertions.assertTrue;
20  
21  import java.util.List;
22  
23  import org.junit.jupiter.api.BeforeEach;
24  import org.junit.jupiter.api.Test;
25  
26  import testdomain.Book;
27  import testdomain.Document;
28  import testdomain.Magazine;
29  import testdomain.PersonDocument;
30  
31  class DiscriminatorTest extends BaseSqlMap {
32  
33    @BeforeEach
34    void setUp() throws Exception {
35      initSqlMap("com/ibatis/sqlmap/maps/SqlMapConfig.xml", null);
36      initScript("scripts/docs-init.sql");
37    }
38  
39    @Test
40    void testDiscriminator() throws Exception {
41  
42      List<?> list = sqlMap.queryForList("getDocuments", null);
43      assertEquals(6, list.size());
44  
45      assertTrue(list.get(0) instanceof Book);
46      assertTrue(list.get(1) instanceof Magazine);
47      assertTrue(list.get(2) instanceof Book);
48      assertTrue(list.get(3) instanceof Magazine);
49      assertTrue(list.get(4) instanceof Document);
50      assertTrue(list.get(5) instanceof Document);
51  
52      assertEquals(1, ((Document) list.get(0)).getId());
53      assertEquals(2, ((Document) list.get(1)).getId());
54      assertEquals(3, ((Document) list.get(2)).getId());
55      assertEquals(4, ((Document) list.get(3)).getId());
56      assertEquals(5, ((Document) list.get(4)).getId());
57      assertEquals(6, ((Document) list.get(5)).getId());
58  
59      assertEquals(Integer.valueOf(55), ((Book) list.get(0)).getPages());
60      assertEquals("Lyon", ((Magazine) list.get(1)).getCity());
61      assertEquals(Integer.valueOf(3587), ((Book) list.get(2)).getPages());
62      assertEquals("Paris", ((Magazine) list.get(3)).getCity());
63    }
64  
65    @Test
66    void testDiscriminatorInNestedResultMap() throws Exception {
67      List<?> list = sqlMap.queryForList("getPersonDocuments");
68      assertEquals(3, list.size());
69  
70      assertTrue(((PersonDocument) list.get(0)).getFavoriteDocument() instanceof Magazine);
71      assertTrue(((PersonDocument) list.get(1)).getFavoriteDocument() instanceof Book);
72      assertTrue(((PersonDocument) list.get(2)).getFavoriteDocument() instanceof Document);
73  
74    }
75  
76    @Test
77    void testDiscriminatorWithNestedResultMap() throws Exception {
78      List<?> list = sqlMap.queryForList("getDocumentsWithAttributes");
79      assertEquals(6, list.size());
80  
81      assertTrue(list.get(0) instanceof Book);
82      Book b = (Book) list.get(0);
83      assertEquals(2, b.getAttributes().size());
84  
85      assertTrue(list.get(1) instanceof Magazine);
86      Magazine m = (Magazine) list.get(1);
87      assertEquals(1, m.getAttributes().size());
88  
89      assertTrue(list.get(2) instanceof Book);
90      b = (Book) list.get(2);
91      assertEquals(2, b.getAttributes().size());
92  
93      Document d = (Document) list.get(3);
94      assertEquals(0, d.getAttributes().size());
95  
96      d = (Document) list.get(4);
97      assertEquals(0, d.getAttributes().size());
98  
99      d = (Document) list.get(5);
100     assertEquals(0, d.getAttributes().size());
101   }
