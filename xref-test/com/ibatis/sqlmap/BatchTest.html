46    void testExecutebatchDetailed() {

57      account.setFirstName("fred");
58      account.setLastName("flintstone");
59      account.setEmailAddress("fred.flintstone@gmail.com");
60      accountList1.add(account);
61  
62      account = new Account();
63      account.setId(12);
64      account.setFirstName("fred");
65      account.setLastName("flintstone");
66      account.setEmailAddress("fred.flintstone@gmail.com");
67      accountList1.add(account);
68  
69      account = new Account();
70      account.setId(13);
71      account.setFirstName("fred");
72      account.setLastName("flintstone");
73      account.setEmailAddress("fred.flintstone@gmail.com");
74      accountList1.add(account);
75  
76      account = new Account();
77      account.setId(14);
78      account.setFirstName("fred");
79      account.setLastName("flintstone");
80      account.setEmailAddress("fred.flintstone@gmail.com");
81      accountList1.add(account);
82  
83      List<Account> accountList2 = new ArrayList<Account>();
84      account = new Account();
85      account.setId(15);
86      account.setFirstName("fred");
87      account.setLastName("flintstone");
88      account.setEmailAddress("fred.flintstone@gmail.com");
89      accountList2.add(account);
90  
91      account = new Account();
92      account.setId(16);
93      account.setFirstName("fred");
94      account.setLastName("flintstone");
95      account.setEmailAddress("fred.flintstone@gmail.com");
96      accountList2.add(account);
97  
98      account = new Account();
99      account.setId(17);
100     account.setFirstName("fred");
101     account.setLastName("flintstone");
102     account.setEmailAddress("fred.flintstone@gmail.com");
103     accountList2.add(account);
104 
105     account = new Account();
106     account.setId(18);
107     account.setFirstName("fred");
108     account.setLastName("flintstone");
109     account.setEmailAddress("fred.flintstone@gmail.com");
110     accountList2.add(account);
111 
112     try {
113       sqlMap.startTransaction();
114       sqlMap.startBatch();
115 
116       // insert 5 accounts
117       for (int i = 0; i < accountList1.size(); i++) {
118         sqlMap.insert("insertAccountViaInlineParameters", accountList1.get(i));
119       }
120 
121       // update 1 account
122       account = new Account();
123       account.setId(10);
124       account.setFirstName("barney");
125       account.setLastName("rubble");
126       account.setEmailAddress("barney.rubble@gmail.com");
127 
128       sqlMap.update("updateAccountViaInlineParameters", account);
129 
130       // insert 4 accounts
131       for (int i = 0; i < accountList2.size(); i++) {
132         sqlMap.insert("insertAccountViaInlineParameters", accountList2.get(i));
133       }
134 
135       List<?> results = sqlMap.executeBatchDetailed();
136       sqlMap.commitTransaction();
137 
138       assertEquals(3, results.size());
139 
140       BatchResult br = (BatchResult) results.get(0);
141       assertEquals(5, br.getUpdateCounts().length);
142 
143       br = (BatchResult) results.get(1);
144       assertEquals(1, br.getUpdateCounts().length);
145 
146       br = (BatchResult) results.get(2);
147       assertEquals(4, br.getUpdateCounts().length);
148 
149     } catch (BatchException e) {
150       fail(e.getMessage());
151     } catch (SQLException e) {
152       fail(e.getMessage());
153     } finally {
154       try {
155         sqlMap.endTransaction();
156       } catch (SQLException e) {
157         fail(e.getMessage());
158       }
159     }
160   }

256       // executed)
257       account = new Account();
258       sqlMap.insert("insertAccountViaInlineParameters", account);
259

261       account = new Account();

272       }
273 
274       sqlMap.executeBatchDetailed();
275       fail("This statement should not get executed - we expect an SQLException");
276     } catch (BatchException e) {
277       // the first statement of the failing batch should have executed OK
278       BatchUpdateException bue = e.getBatchUpdateException();
279       assertEquals(1, bue.getUpdateCounts().length);
280 
281       List<?> results = e.getSuccessfulBatchResults();
282       assertEquals(2, results.size());
283       BatchResult br = (BatchResult) results.get(0);
284       assertEquals(5, br.getUpdateCounts().length);
285       br = (BatchResult) results.get(1);
286       assertEquals(1, br.getUpdateCounts().length);
287     } catch (SQLException e) {
288       fail(e.getMessage());
289     } finally {
290       try {
291         sqlMap.endTransaction();
292       } catch (SQLException e) {
293         fail(e.getMessage());
294       }
295     }
