16  package com.ibatis.sqlmap;
17  
18  import com.ibatis.common.beans.Probe;
19  import com.ibatis.common.beans.ProbeFactory;
20  import org.w3c.dom.Document;
21  
22  import javax.xml.parsers.DocumentBuilderFactory;
23  import javax.xml.parsers.ParserConfigurationException;
24  import java.sql.SQLException;
25  
26  public class DomStatementTest extends BaseSqlMapTest {
27  
28    // SETUP & TEARDOWN
29  
30    @Override
31    protected void setUp() throws Exception {
32      initSqlMap("com/ibatis/sqlmap/maps/SqlMapConfig.xml", null);
33      initScript("scripts/account-init.sql");
34      initScript("scripts/order-init.sql");
35      initScript("scripts/line_item-init.sql");
36    }
37  
38    @Override
39    protected void tearDown() throws Exception {
40    }
41  
42    public void testExecuteQueryForDom() throws SQLException {
43      Document account = (Document) sqlMap.queryForObject("getAccountDom", newParameter("1"));
44      assertNotNull(account);
45  
46      Probe dom = ProbeFactory.getProbe(account);
47  
48      assertEquals("1", dom.getObject(account, "ID"));
49      assertEquals("Clinton", dom.getObject(account, "FIRSTNAME"));
50      assertEquals("Begin", dom.getObject(account, "LASTNAME"));
51      assertEquals("clinton.begin@ibatis.com", dom.getObject(account, "EMAILADDRESS"));
52    }
53  
54    public void testExecuteQueryForDomSpecialChars() throws SQLException {
55      Document account = (Document) sqlMap.queryForObject("getAccountDom", newParameter("5"));
56      assertNotNull(account);
57  
58      Probe dom = ProbeFactory.getProbe(account);
59  
60      assertEquals("5", dom.getObject(account, "ID"));
61      assertEquals("&manda", dom.getObject(account, "FIRSTNAME"));
62    }
63  
64    public void testExecuteQueryForDomExternalMaps() throws SQLException {
65      Document account = (Document) sqlMap.queryForObject("getAccountDomExternalMaps", newParameter("1"));
66      assertNotNull(account);
67  
68      Probe dom = ProbeFactory.getProbe(account);
69  
70      assertEquals("1", dom.getObject(account, "id"));
71      assertEquals("Clinton", dom.getObject(account, "firstName"));
72      assertEquals("Begin", dom.getObject(account, "lastName"));
73      assertEquals("clinton.begin@ibatis.com", dom.getObject(account, "emailAddress"));
74      assertEquals("1", dom.getObject(account, "account.ID"));
75      assertEquals("Clinton", dom.getObject(account, "account.FIRSTNAME"));
76      assertEquals("Begin", dom.getObject(account, "account.LASTNAME"));
77      assertEquals("clinton.begin@ibatis.com", dom.getObject(account, "account.EMAILADDRESS"));
78    }
79  
80    public void testExecuteQueryForOrderDom() throws SQLException {
81  
82      Document order = (Document) sqlMap.queryForObject("getOrderDom", newParameter("1"));
83      assertNotNull(order);
84  
85      Probe dom = ProbeFactory.getProbe(order);
86  
87      assertEquals("1", dom.getObject(order, "id"));
88      assertEquals("2", dom.getObject(order, "lineItems.lineItem[1].ID"));
89    }
90  
91    private Document newParameter(String val) {
92      Document param = newDocument("parameter");
93      Probe dom = ProbeFactory.getProbe(param);
94      dom.setObject(param, "id", val);
95      return param;
96    }
97  
98    private Document newDocument(String root) {
99      try {
100       Document doc = DocumentBuilderFactory.newInstance().newDocumentBuilder().newDocument();
101       doc.appendChild(doc.createElement(root));
102       return doc;
103     } catch (ParserConfigurationException e) {
104       throw new RuntimeException("Error creating XML document.  Cause: " + e);
105     }
106   }
