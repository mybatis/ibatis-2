24  /**
25   * Exercises the indexed-property paths in {@link BaseProbe} (via {@link GenericProbe}) for every array and List type.
26   */
27  class BaseProbeIndexedTest {
28  
29    private final Probe probe = ProbeFactory.getProbe(new Object());
30  
31    // Helper bean that holds every supported array type
32    public static class ArrayBean {
33      public Object[] objects = new Object[] { "a", "b", "c" };
34      public char[] chars = new char[] { 'x', 'y', 'z' };
35      public boolean[] booleans = new boolean[] { true, false, true };
36      public byte[] bytes = new byte[] { 1, 2, 3 };
37      public double[] doubles = new double[] { 1.0, 2.0, 3.0 };
38      public float[] floats = new float[] { 1.0f, 2.0f, 3.0f };
39      public int[] ints = new int[] { 10, 20, 30 };
40      public long[] longs = new long[] { 100L, 200L, 300L };
41      public short[] shorts = new short[] { 4, 5, 6 };
42      public List<String> strings = new ArrayList<>(List.of("one", "two", "three"));
43  
44      public Object[] getObjects() {
45        return objects;
46      }
47  
48      public char[] getChars() {
49        return chars;
50      }
51  
52      public boolean[] getBooleans() {
53        return booleans;
54      }
55  
56      public byte[] getBytes() {
57        return bytes;
58      }
59  
60      public double[] getDoubles() {
61        return doubles;
62      }
63  
64      public float[] getFloats() {
65        return floats;
66      }
67  
68      public int[] getInts() {
69        return ints;
70      }
71  
72      public long[] getLongs() {
73        return longs;
74      }
75  
76      public short[] getShorts() {
77        return shorts;
78      }
79  
80      public List<String> getStrings() {
81        return strings;
82      }
83    }
84

86  
87    @Test
88    void shouldGetFromObjectArray() {
89      ArrayBean bean = new ArrayBean();
90      Assertions.assertEquals("b", probe.getObject(bean, "objects[1]"));
91    }

93    @Test
94    void shouldGetFromCharArray() {
95      ArrayBean bean = new ArrayBean();
96      Assertions.assertEquals(Character.valueOf('y'), probe.getObject(bean, "chars[1]"));
