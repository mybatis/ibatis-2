16  package com.ibatis.common.beans;
17  
18  import static org.junit.jupiter.api.Assertions.assertEquals;
19  
20  import com.ibatis.sqlmap.engine.accessplan.AccessPlan;
21  import com.ibatis.sqlmap.engine.accessplan.AccessPlanFactory;
22  
23  import java.math.BigDecimal;
24  
25  import org.junit.jupiter.api.Test;
26  
27  import testdomain.Order;
28  
29  class PropertyAccessPlanTest {
30  
31    private static final String[] properties = { "id", "id", "account.firstName", "account.lastName",
32        "account.emailAddress", "cardType", "cardNumber", "cardExpiry", "favouriteLineItem.itemCode",
33        "favouriteLineItem.quantity", "favouriteLineItem.price" };
34  
35    private static final Object[] values = { Integer.valueOf(100), Integer.valueOf(100), "Clinton", "Begin",
36        "clinton@ibatis.com", "VISA", "1234567890", "05/06", "M100", Integer.valueOf(3), new BigDecimal(150) };
37  
38    @Test
39    void testSetAndGetProperties() {
40  
41      AccessPlan plan = AccessPlanFactory.getAccessPlan(Order.class, properties);
42  
43      Order order = new Order();
44  
45      plan.setProperties(order, values);
46      assertOrder(order);
47  
48      Object[] newValues = plan.getProperties(order);
49  
50      order = new Order();
51      plan.setProperties(order, newValues);
52  
53      assertOrder(order);
54  
55    }
56  
57    private void assertOrder(Order order) {
58      assertEquals(values[0], Integer.valueOf(order.getId()));
59      assertEquals(values[1], Integer.valueOf(order.getId()));
60      assertEquals(values[2], order.getAccount().getFirstName());
61      assertEquals(values[3], order.getAccount().getLastName());
62      assertEquals(values[4], order.getAccount().getEmailAddress());
63      assertEquals(values[5], order.getCardType());
64      assertEquals(values[6], order.getCardNumber());
65      assertEquals(values[7], order.getCardExpiry());
66      assertEquals(values[8], order.getFavouriteLineItem().getItemCode());
67      assertEquals(values[9], Integer.valueOf(order.getFavouriteLineItem().getQuantity()));
68      assertEquals(values[10], order.getFavouriteLineItem().getPrice());
69    }
70  
71  }
