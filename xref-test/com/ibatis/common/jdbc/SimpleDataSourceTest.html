81    @Test
82    void shouldReadBackAllConfiguredProperties() {
83      Map<String, String> props = baseProps();
84      props.put("Pool.MaximumActiveConnections", "3");
85      props.put("Pool.MaximumIdleConnections", "2");
86      props.put("Pool.MaximumCheckoutTime", "1000");
87      props.put("Pool.TimeToWait", "500");
88      props.put("Pool.PingEnabled", "false");
89      props.put("Pool.PingQuery", "SELECT 1 FROM INFORMATION_SCHEMA.SYSTEM_USERS");
90      props.put("Pool.PingConnectionsOlderThan", "60000");
91      props.put("Pool.PingConnectionsNotUsedFor", "30000");
92      props.put("JDBC.DefaultAutoCommit", "false");
93  
94      SimpleDataSource ds = new SimpleDataSource(props);
95  
96      Assertions.assertEquals(DRIVER, ds.getJdbcDriver());
97      Assertions.assertEquals(URL, ds.getJdbcUrl());
98      Assertions.assertEquals(USER, ds.getJdbcUsername());
99      Assertions.assertEquals(PASS, ds.getJdbcPassword());
100     Assertions.assertEquals(3, ds.getPoolMaximumActiveConnections());
101     Assertions.assertEquals(2, ds.getPoolMaximumIdleConnections());
102     Assertions.assertEquals(1000, ds.getPoolMaximumCheckoutTime());
103     Assertions.assertEquals(500, ds.getPoolTimeToWait());
104     Assertions.assertFalse(ds.isPoolPingEnabled());
105     Assertions.assertEquals("SELECT 1 FROM INFORMATION_SCHEMA.SYSTEM_USERS", ds.getPoolPingQuery());
106     Assertions.assertEquals(60000, ds.getPoolPingConnectionsOlderThan());
107     Assertions.assertEquals(30000, ds.getPoolPingConnectionsNotUsedFor());

239   @Test
240   void shouldUnwrapPooledConnection() throws Exception {
241     SimpleDataSource ds = new SimpleDataSource(baseProps());
242     Connection proxy = ds.getConnection();
243     // The proxy is a JDK Proxy (implements Connection) wrapping a SimplePooledConnection handler.
244     // unwrapConnection checks instanceof SimplePooledConnection which is false for the JDK proxy,
245     // so it returns the same object unchanged.
246     Connection result = SimpleDataSource.unwrapConnection(proxy);
247     Assertions.assertSame(proxy, result);
248 
249     // Verify the actual unwrap path: if we pass a real connection it comes back unchanged.
250     SimpleDataSource.SimplePooledConnection spc = (SimpleDataSource.SimplePooledConnection) java.lang.reflect.Proxy
251         .getInvocationHandler(proxy);
252     Connection realConn = spc.getRealConnection();
253     Connection realResult = SimpleDataSource.unwrapConnection(realConn);
254     Assertions.assertSame(realConn, realResult);
255 
256     proxy.close();
257   }

275   @Test
276   void shouldRecordTimestampsOnPooledConnection() throws Exception {
277     SimpleDataSource ds = new SimpleDataSource(baseProps());
278     Connection proxy = ds.getConnection();
279 
280     // Reach the internal SimplePooledConnection via unwrapConnection
281     // (proxy implements Connection but its handler is SimplePooledConnection)
282     java.lang.reflect.InvocationHandler handler = java.lang.reflect.Proxy.getInvocationHandler(proxy);
283     Assertions.assertInstanceOf(SimpleDataSource.SimplePooledConnection.class, handler);
284 
285     SimpleDataSource.SimplePooledConnection spc = (SimpleDataSource.SimplePooledConnection) handler;
286 
287     Assertions.assertTrue(spc.getCreatedTimestamp() > 0);
288     Assertions.assertTrue(spc.getCheckoutTimestamp() > 0);
289     Assertions.assertTrue(spc.getLastUsedTimestamp() > 0);
290     Assertions.assertTrue(spc.getAge() >= 0);
291     Assertions.assertTrue(spc.getTimeElapsedSinceLastUse() >= 0);
292     Assertions.assertTrue(spc.getCheckoutTime() >= 0);
293     Assertions.assertNotNull(spc.getRealConnection());
294     Assertions.assertNotNull(spc.getProxyConnection());
295     Assertions.assertTrue(spc.getRealHashCode() != 0);
296 
297     // setter round-trips
298     spc.setCreatedTimestamp(1000L);
299     Assertions.assertEquals(1000L, spc.getCreatedTimestamp());
300     spc.setLastUsedTimestamp(2000L);
301     Assertions.assertEquals(2000L, spc.getLastUsedTimestamp());
302     spc.setCheckoutTimestamp(3000L);
303     Assertions.assertEquals(3000L, spc.getCheckoutTimestamp());
304     spc.setConnectionTypeCode(42);
305     Assertions.assertEquals(42, spc.getConnectionTypeCode());
306 
307     // hashCode / equals contracts
308     Assertions.assertEquals(spc.hashCode(), spc.hashCode());
309     Assertions.assertTrue(spc.equals(spc.getRealConnection()));
310 
311     proxy.close();
312   }
