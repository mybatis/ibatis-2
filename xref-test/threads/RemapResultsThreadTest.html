16  package threads;
17  
18  import static org.junit.jupiter.api.Assertions.assertTrue;
19  
20  import com.ibatis.common.resources.Resources;
21  import com.ibatis.sqlmap.client.SqlMapClient;
22  import com.ibatis.sqlmap.client.SqlMapClientBuilder;
23  
24  import java.io.IOException;
25  import java.io.Reader;
26  import java.sql.SQLException;
27  import java.util.Date;
28  import java.util.LinkedList;
29  import java.util.List;
30  
31  import org.junit.jupiter.api.Test;
32  
33  class RemapResultsThreadTest {
34  
35    @Test
36    void testWithRemap() throws Exception {
37      runTest("WithRemap");
38    }
39  
40    @Test
41    void testWithoutRemap() throws Exception {
42      runTest("WithoutRemap");
43    }
44  
45    private void runTest(String statementToRun) throws IOException, SQLException {
46      String resource = "threads/sql-map-config.xml";
47      Reader reader = Resources.getResourceAsReader(resource);
48      SqlMapClient sqlMap = SqlMapClientBuilder.buildSqlMapClient(reader);
49  
50      sqlMap.update("drop");
51      sqlMap.update("create");
52  
53      for (int i = 0; i < 100; i++) {
54        sqlMap.update("inserta");
55        sqlMap.update("insertb");
56        sqlMap.update("insertc");
57      }
58  
59      int count = 2;
60  
61      List<MyThread> threads = new LinkedList<MyThread>();
62  
63      for (int i = 0; i < count; i++) {
64        MyThread thread = new MyThread(sqlMap, statementToRun);
65        thread.start();
66        threads.add(thread);
67      }
68  
69      Date d1 = new Date(new Date().getTime() + 10000);
70  
71      // let's do the test for 10 seconds - for me it failed quite early
72      while (new Date().before(d1)) {
73        for (MyThread myThread : threads) {
74          assertTrue(myThread.isAlive());
75        }
76      }
77    }
78  
79  }
