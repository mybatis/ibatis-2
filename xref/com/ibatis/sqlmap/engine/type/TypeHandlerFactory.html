26  public class TypeHandlerFactory {

28    /** The type handler map. */
29    private final Map typeHandlerMap = new HashMap();

31    /** The unknown type handler. */
32    private final TypeHandler unknownTypeHandler = new UnknownTypeHandler(this);
33  
34    /** The type aliases. */
35    private final HashMap typeAliases = new HashMap();
36  
37    /** The Constant reversePrimitiveMap. */
38    private static final Map reversePrimitiveMap = new HashMap() {
39      {
40        put(Byte.class, byte.class);
41        put(Short.class, short.class);
42        put(Integer.class, int.class);
43        put(Long.class, long.class);
44        put(Float.class, float.class);
45        put(Double.class, double.class);
46        put(Boolean.class, boolean.class);
47      }
48    };

129   /* Public Methods */

131   /**
132    * Get a TypeHandler for a class.
133    *
134    * @param type
135    *          - the class you want a TypeHandler for
136    *
137    * @return - the handler
138    */
139   public TypeHandler getTypeHandler(Class type) {
140     return getTypeHandler(type, null);
141   }
142 
143   /**
144    * Get a TypeHandler for a class and a JDBC type.
145    *
146    * @param type
147    *          - the class
148    * @param jdbcType
149    *          - the jdbc type
150    *
151    * @return - the handler
152    */
153   public TypeHandler getTypeHandler(Class type, String jdbcType) {
154     Map jdbcHandlerMap = (Map) typeHandlerMap.get(type);
155     TypeHandler handler = null;
156     if (jdbcHandlerMap != null) {
157       handler = (TypeHandler) jdbcHandlerMap.get(jdbcType);
158       if (handler == null) {
159         handler = (TypeHandler) jdbcHandlerMap.get(null);
160       }
161     }
162     if (handler == null && type != null && Enum.class.isAssignableFrom(type)) {
163       handler = new EnumTypeHandler(type);
164     }
165     return handler;
166   }
167 
168   /**
169    * When in doubt, get the "unknown" type handler.
170    *
171    * @return - if I told you, it would not be unknown, would it?
172    */
173   public TypeHandler getUnkownTypeHandler() {
174     return unknownTypeHandler;
175   }
176 
177   /**
178    * Tells you if a particular class has a TypeHandler.
179    *
180    * @param type
181    *          - the class
182    *
183    * @return - true if there is a TypeHandler
184    */
185   public boolean hasTypeHandler(Class type) {
186     return type != null && (getTypeHandler(type) != null || Enum.class.isAssignableFrom(type));
187   }

189   /**
190    * Register (add) a type handler for a class.
191    *
192    * @param type
193    *          - the class
194    * @param handler
195    *          - the handler instance
196    */
197   public void register(Class type, TypeHandler handler) {
198     register(type, null, handler);
199   }
200 
201   /**
202    * Register (add) a type handler for a class and JDBC type.
203    *
204    * @param type
205    *          - the class
206    * @param jdbcType
207    *          - the JDBC type
208    * @param handler
209    *          - the handler instance
210    */
211   public void register(Class type, String jdbcType, TypeHandler handler) {
212     Map map = (Map) typeHandlerMap.get(type);
213     if (map == null) {
214       map = new HashMap();
215       typeHandlerMap.put(type, map);
216     }
217     map.put(jdbcType, handler);
218 
219     if (reversePrimitiveMap.containsKey(type)) {
220       register((Class) reversePrimitiveMap.get(type), jdbcType, handler);
221     }
222   }
223 
224   /**
225    * Lookup an aliased class and return it's REAL name.
226    *
227    * @param string
228    *          - the alias
229    *
230    * @return - the REAL name
231    */
232   public String resolveAlias(String string) {
233     String key = null;
234     if (string != null)
235       key = string.toLowerCase();
236     String value = null;
237     if (typeAliases.containsKey(key)) {
238       value = (String) typeAliases.get(key);
239     } else {
240       value = string;
241     }
242 
243     return value;
244   }
245 
246   /**
247    * Adds a type alias that is case insensitive. All of the following String, string, StRiNg will equate to the same
248    * alias.
249    *
250    * @param alias
251    *          - the alias
252    * @param value
253    *          - the real class name
254    */
255   public void putTypeAlias(String alias, String value) {
256     String key = null;
257     if (alias != null)
258       key = alias.toLowerCase();
259     if (typeAliases.containsKey(key) && !typeAliases.get(key).equals(value)) {
260       throw new SqlMapException("Error in XmlSqlMapClientBuilder.  Alias name conflict occurred.  The alias '" + key
261           + "' is already mapped to the value '" + typeAliases.get(alias) + "'.");
262     }
263     typeAliases.put(key, value);
264   }
