16  package com.ibatis.sqlmap.engine.accessplan;
17  
18  import java.util.Map;
19  
20  /**
21   * Factory to get an accesss plan appropriate for an object.
22   */
23  public class AccessPlanFactory {
24  
25    /** The bytecode enhancement enabled. */
26    private static boolean bytecodeEnhancementEnabled = false;
27  
28    /**
29     * Instantiates a new access plan factory.
30     */
31    private AccessPlanFactory() {
32    }
33  
34    /**
35     * Creates an access plan for working with a bean.
36     *
37     * @param clazz
38     *          the clazz
39     * @param propertyNames
40     *          the property names
41     *
42     * @return An access plan
43     */
44    public static AccessPlan getAccessPlan(Class clazz, String[] propertyNames) {
45      AccessPlan plan;
46  
47      boolean complex = false;
48  
49      if (clazz == null || propertyNames == null) {
50        complex = true;
51      } else {
52        for (int i = 0; i < propertyNames.length; i++) {
53          if (propertyNames[i].indexOf('[') > -1 || propertyNames[i].indexOf('.') > -1) {
54            complex = true;
55            break;
56          }
57        }
58      }
59  
60      if (complex) {
61        plan = new ComplexAccessPlan(clazz, propertyNames);
62      } else if (Map.class.isAssignableFrom(clazz)) {
63        plan = new MapAccessPlan(clazz, propertyNames);
64      } else {
65        // Possibly causes bug 945746 --but the bug is unconfirmed (can't be reproduced)
66        if (bytecodeEnhancementEnabled) {
67          try {
68            plan = new EnhancedPropertyAccessPlan(clazz, propertyNames);
69          } catch (Throwable t) {
70            try {
71              plan = new PropertyAccessPlan(clazz, propertyNames);
72            } catch (Throwable t2) {
73              plan = new ComplexAccessPlan(clazz, propertyNames);
74            }
75          }
76        } else {
77          try {
78            plan = new PropertyAccessPlan(clazz, propertyNames);
79          } catch (Throwable t) {
80            plan = new ComplexAccessPlan(clazz, propertyNames);
81          }
82        }
83      }
84      return plan;
85    }
86  
87    /**
88     * Tells whether or not bytecode enhancement (CGLIB, etc) is enabled.
89     *
90     * @return true if bytecode enhancement is enabled
91     */
92    public static boolean isBytecodeEnhancementEnabled() {
93      return bytecodeEnhancementEnabled;
94    }
95  
96    /**
97     * Turns on or off bytecode enhancement (CGLIB, etc).
98     *
99     * @param bytecodeEnhancementEnabled
100    *          - the switch
101    */
102   public static void setBytecodeEnhancementEnabled(boolean bytecodeEnhancementEnabled) {
103     AccessPlanFactory.bytecodeEnhancementEnabled = bytecodeEnhancementEnabled;
104   }
105 
106 }
