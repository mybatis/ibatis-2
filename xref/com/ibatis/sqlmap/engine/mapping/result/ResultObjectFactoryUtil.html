16  package com.ibatis.sqlmap.engine.mapping.result;
17  
18  import com.ibatis.common.resources.Resources;
19  
20  import java.util.*;
21  
22  /**
23   * This class is used to create instances of result objects. It will use the configured ResultObjectFactory if there is
24   * one, otherwise it will use iBATIS' normal methods.
25   * <p>
26   * Note that this class is somewhat tightly coupled with SqlExecuter - SqlExecute must call the setStatementId() and
27   * setResultObjectFactory() methods before executing a statement. This is a result of using a ThreadLocal to hold the
28   * current configuration for the statement under execution. Using a ThreadLocal is a solution for IBATIS-366. Without a
29   * ThreadLocal, the current factory and statement id would have to be added to many method signatures - often in
30   * inappropriate places.
31   *
32   * @author Jeff Butler
33   */
34  public class ResultObjectFactoryUtil {
35  
36    /**
37     * Use a ThreadLocal to hold the current statementId and factory. This is much easier than passing these items all
38     * over the place, and it has no measurable impact on performance (I did a test with 100000 result rows and found no
39     * impact - Jeff Butler).
40     */
41    private static ThreadLocal<Deque<FactorySettings>> factorySettings = new ThreadLocal<>();
42  
43    /**
44     * Utility class - no instances.
45     */
46    private ResultObjectFactoryUtil() {
47      super();
48    }
49  
50    /**
51     * Algorithm:
52     * <ul>
53     * <li>If factory is null, then create object internally()</li>
54     * <li>Otherwise try to create object through factory</li>
55     * <li>If null returned from factory, then create object internally</li>
56     * </ul>
57     * This allows the factory to selectively create objects, also allows for the common possibility that a factory is not
58     * configured.
59     *
60     * @param clazz
61     *          the type of object to create
62     *
63     * @return a new instance of the specified class. The instance must be castable to the specified class.
64     *
65     * @throws InstantiationException
66     *           if the instance cannot be created. If you throw this Exception, iBATIS will throw a runtime exception in
67     *           response and will end.
68     * @throws IllegalAccessException
69     *           if the constructor cannot be accessed. If you throw this Exception, iBATIS will throw a runtime exception
70     *           in response and will end.
71     */
72    public static Object createObjectThroughFactory(Class clazz) throws InstantiationException, IllegalAccessException {
73  
74      FactorySettings fs = getCurrentFactorySettings();
75  
76      Object obj;
77      if (fs.getResultObjectFactory() == null) {
78        obj = createObjectInternally(clazz);
79      } else {
80        obj = fs.getResultObjectFactory().createInstance(fs.getStatementId(), clazz);
81        if (obj == null) {
82          obj = createObjectInternally(clazz);
83        }
84      }
85  
86      return obj;
87    }
88  
89    /**
90     * This method creates object using iBATIS' normal mechanism. We translate List and Collection to ArrayList, and Set
91     * to HashSet because these interfaces may be requested in nested resultMaps and we want to supply default
92     * implementations.
93     *
94     * @param clazz
95     *          the clazz
96     *
97     * @return the object
98     *
99     * @throws InstantiationException
100    *           the instantiation exception
101    * @throws IllegalAccessException
102    *           the illegal access exception
103    */
104   private static Object createObjectInternally(Class clazz) throws InstantiationException, IllegalAccessException {
105     Class classToCreate;
106     if (clazz == List.class || clazz == Collection.class) {
107       classToCreate = ArrayList.class;
108     } else if (clazz == Set.class) {
109       classToCreate = HashSet.class;
110     } else {
111       classToCreate = clazz;
112     }
113 
114     return Resources.instantiate(classToCreate);
115   }
116 
117   /**
118    * This method pushes a new result object factory configuration onto the stack. We use a stack because the method can
119    * be called in a "nested" fashion if there are sub-selects. Calls to this method should be equally balanced with
120    * calls to cleanupResultObjectFactory().
121    *
122    * @param resultObjectFactory
123    *          the result object factory
124    * @param statementId
125    *          the statement id
126    */
127   public static void setupResultObjectFactory(ResultObjectFactory resultObjectFactory, String statementId) {
128     Deque<FactorySettings> fss = factorySettings.get();
129     if (fss == null) {
130       fss = new ArrayDeque<>();
131       factorySettings.set(fss);
132     }
133 
134     FactorySettings fs = new FactorySettings();
135     fs.setResultObjectFactory(resultObjectFactory);
136     fs.setStatementId(statementId);
137     fss.push(fs);
138   }
139 
140   /**
141    * Removes the FactorySettings bound to the current thread to avoid classloader leak issues. This method pops the top
142    * item off the stack, and kills the stack if there are no items left.
143    */
144   public static void cleanupResultObjectFactory() {
145     Deque<FactorySettings> fss = factorySettings.get();
146     if (!fss.isEmpty()) {
147       fss.pop();
148     }
149 
150     if (fss.isEmpty()) {
151       factorySettings.remove();
152     }
153   }
154 
155   /**
156    * Gets the current factory settings.
157    *
158    * @return the current factory settings
159    */
160   private static FactorySettings getCurrentFactorySettings() {
161     Deque<FactorySettings> fss = factorySettings.get();
162     FactorySettings fs;
163     if (fss == null || fss.isEmpty()) {
164       // this shouldn't happen if the SqlExecuter is behaving correctly
165       fs = new FactorySettings();
166     } else {
167       fs = fss.peek();
168     }
169 
170     return fs;
171   }
172 
173   /**
174    * The Class FactorySettings.
175    */
176   private static class FactorySettings {
177 
178     /** The result object factory. */
179     private ResultObjectFactory resultObjectFactory;
180 
181     /** The statement id. */
182     private String statementId;
183 
184     /**
185      * Gets the result object factory.
186      *
187      * @return the result object factory
188      */
189     public ResultObjectFactory getResultObjectFactory() {
190       return resultObjectFactory;
191     }
192 
193     /**
194      * Sets the result object factory.
195      *
196      * @param resultObjectFactory
197      *          the new result object factory
198      */
199     public void setResultObjectFactory(ResultObjectFactory resultObjectFactory) {
200       this.resultObjectFactory = resultObjectFactory;
201     }
202 
203     /**
204      * Gets the statement id.
205      *
206      * @return the statement id
207      */
208     public String getStatementId() {
209       return statementId;
210     }
211 
212     /**
213      * Sets the statement id.
214      *
215      * @param statementId
216      *          the new statement id
217      */
218     public void setStatementId(String statementId) {
219       this.statementId = statementId;
220     }
221   }
222 }
