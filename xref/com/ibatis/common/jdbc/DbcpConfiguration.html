16  package com.ibatis.common.jdbc;
17  
18  import com.ibatis.common.beans.Probe;
19  import com.ibatis.common.beans.ProbeFactory;
20  
21  import java.util.Iterator;
22  import java.util.Map;
23  
24  import javax.sql.DataSource;
25  
26  import org.apache.commons.dbcp.BasicDataSource;
27  
28  /**
29   * Wrapper class to simplify use of DBCP.
30   */
31  public class DbcpConfiguration {
32  
33    /** The Constant PROBE. */
34    private static final Probe PROBE = ProbeFactory.getProbe();
35  
36    /** The Constant ADD_DRIVER_PROPS_PREFIX. */
37    private static final String ADD_DRIVER_PROPS_PREFIX = "Driver.";
38  
39    /** The Constant ADD_DRIVER_PROPS_PREFIX_LENGTH. */
40    private static final int ADD_DRIVER_PROPS_PREFIX_LENGTH = ADD_DRIVER_PROPS_PREFIX.length();
41  
42    /** The data source. */
43    private DataSource dataSource;
44  
45    /**
46     * Constructor to supply a map of properties.
47     *
48     * @param properties
49     *          - the map of configuration properties
50     */
51    public DbcpConfiguration(Map properties) {
52      try {
53  
54        dataSource = legacyDbcpConfiguration(properties);
55        if (dataSource == null) {
56          dataSource = newDbcpConfiguration(properties);
57        }
58  
59      } catch (Exception e) {
60        throw new RuntimeException("Error initializing DbcpDataSourceFactory.  Cause: " + e, e);
61      }
62    }
63  
64    /**
65     * Getter for DataSource.
66     *
67     * @return The DataSource
68     */
69    public DataSource getDataSource() {
70      return dataSource;
71    }
72  
73    /**
74     * New dbcp configuration.
75     *
76     * @param map
77     *          the map
78     *
79     * @return the basic data source
80     */
81    private BasicDataSource newDbcpConfiguration(Map map) {
82      BasicDataSource basicDataSource = new BasicDataSource();
83      Iterator props = map.keySet().iterator();
84      while (props.hasNext()) {
85        String propertyName = (String) props.next();
86        if (propertyName.startsWith(ADD_DRIVER_PROPS_PREFIX)) {
87          String value = (String) map.get(propertyName);
88          basicDataSource.addConnectionProperty(propertyName.substring(ADD_DRIVER_PROPS_PREFIX_LENGTH), value);
89        } else if (PROBE.hasWritableProperty(basicDataSource, propertyName)) {
90          String value = (String) map.get(propertyName);
91          Object convertedValue = convertValue(basicDataSource, propertyName, value);
92          PROBE.setObject(basicDataSource, propertyName, convertedValue);
93        }
94      }
95      return basicDataSource;
96    }
97  
98    /**
99     * Convert value.
100    *
101    * @param object
102    *          the object
103    * @param propertyName
104    *          the property name
105    * @param value
106    *          the value
107    *
108    * @return the object
109    */
110   private Object convertValue(Object object, String propertyName, String value) {
111     Object convertedValue = value;
112     Class targetType = PROBE.getPropertyTypeForSetter(object, propertyName);
113     if (targetType == Integer.class || targetType == int.class) {
114       convertedValue = Integer.valueOf(value);
115     } else if (targetType == Long.class || targetType == long.class) {
116       convertedValue = Long.valueOf(value);
117     } else if (targetType == Boolean.class || targetType == boolean.class) {
118       convertedValue = Boolean.valueOf(value);
119     }
120     return convertedValue;
121   }
122 
123   /**
124    * Legacy dbcp configuration.
125    *
126    * @param map
127    *          the map
128    *
129    * @return the basic data source
130    */
131   private BasicDataSource legacyDbcpConfiguration(Map map) {
132     BasicDataSource basicDataSource = null;
133     if (map.containsKey("JDBC.Driver")) {
134       basicDataSource = new BasicDataSource();
135       String driver = (String) map.get("JDBC.Driver");
136       String url = (String) map.get("JDBC.ConnectionURL");
137       String username = (String) map.get("JDBC.Username");
138       String password = (String) map.get("JDBC.Password");
139       String validationQuery = (String) map.get("Pool.ValidationQuery");
140       String maxActive = (String) map.get("Pool.MaximumActiveConnections");
141       String maxIdle = (String) map.get("Pool.MaximumIdleConnections");
142       String maxWait = (String) map.get("Pool.MaximumWait");
143 
144       basicDataSource.setUrl(url);
145       basicDataSource.setDriverClassName(driver);
146       basicDataSource.setUsername(username);
147       basicDataSource.setPassword(password);
148 
149       if (notEmpty(validationQuery)) {
150         basicDataSource.setValidationQuery(validationQuery);
151       }
152 
153       if (notEmpty(maxActive)) {
154         basicDataSource.setMaxActive(Integer.parseInt(maxActive));
155       }
156 
157       if (notEmpty(maxIdle)) {
158         basicDataSource.setMaxIdle(Integer.parseInt(maxIdle));
159       }
160 
161       if (notEmpty(maxWait)) {
162         basicDataSource.setMaxWait(Integer.parseInt(maxWait));
163       }
164 
165       Iterator props = map.keySet().iterator();
166       while (props.hasNext()) {
167         String propertyName = (String) props.next();
168         if (propertyName.startsWith(ADD_DRIVER_PROPS_PREFIX)) {
169           String value = (String) map.get(propertyName);
170           basicDataSource.addConnectionProperty(propertyName.substring(ADD_DRIVER_PROPS_PREFIX_LENGTH), value);
171         }
172       }
173     }
174     return basicDataSource;
175   }
176 
177   /**
178    * Not empty.
179    *
180    * @param s
181    *          the s
182    *
183    * @return true, if successful
184    */
185   private boolean notEmpty(String s) {
186     return s != null && s.length() > 0;
187   }
188 
189 }
